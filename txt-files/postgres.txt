CREATE DATABASE course;

CREATE TABLE "books"
(
  "id" SERIAL,
  "title" VARCHAR(100),
  "year" INT,
  "author" VARCHAR(100),
  "price" INT
);

CREATE TABLE "publishers"
(
  "id" SERIAL,
  "name" VARCHAR(100),
  "site" VARCHAR(100)
);
